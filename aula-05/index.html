Inserindo um longo código em python: 

import numpy as np
        import pandas as pd
        
        # Exemplo de conjunto de dados
        data = {'valor': [10, 12, 14, 15, 100, 18, 19, 20, 21, 22, 23, 24, 25, 30, 32]}
        df = pd.DataFrame(data)
        
        # Calcular os quartis e o IQR
        Q1 = df['valor'].quantile(0.25)
        Q3 = df['valor'].quantile(0.75)
        IQR = Q3 - Q1
        
        # Definir os limites inferior e superior
        limite_inferior = Q1 - 1.5 * IQR
        limite_superior = Q3 + 1.5 * IQR
        
        # Remover os outliers
        df_sem_outliers = df[(df['valor'] >= limite_inferior) & (df['valor'] <= limite_superior)]
        
        print("Dados originais:")
        print(df)
        print("\nDados após remoção dos outliers:")
        print(df_sem_outliers)
